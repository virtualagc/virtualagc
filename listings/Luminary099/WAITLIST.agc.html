Source Code

These source-code files were from the printout at the MIT Museum, with many thanks to Deborah Douglas, the Museum's Curator of Science and Technology, for releasing it to us. The printout was digitally photographed by Paul Fjeld. A team of volunteers transcribed the source code manually from these images or, where appropriate, modified already-transcribed but similar source Luminary 131 (Apollo 13) source code to agree with the images from the printout. Note that the page images presented online are of reduced quality, and that higher-quality images are available. Report any conversion errors or legibility problems in page images to info@sandroid.org. Notations on the program listing read, in part:

	GAP:  ASSEMBLE REVISION 001 OF AGC PROGRAM LMY99 BY NASA 2021112-061
	20'35 OCT. 28,1968
Note that the date is the date of the printout, not the date of the program revision.

047779,000047:                                                                                                  #  FUNCTIONAL DESCRIPTION -
047780,000048:                                                                                                  #        PART OF A SECTION OF PROGRAMS,- WAITLIST, TASKOVER, T3RUPT, USED TO CALL A PROGRAM, (CALLED A TASK),
047781,000049:                                                                                                  #        WHICH IS TO BEGIN IN C(A) CENTISECONDS.  WAITLIST UPDATES TIME3, LST1 AND LST2.  THE MEANING OF THESE LISTS
047782,000050:                                                                                                  #        FOLLOW.
047783,000051: 
047784,000052:                                                                                                  #                C(TIME3) = 16384 -(T1-T) CENTISECONDS, (T=PRESENT TIME, T1-TIME FOR TASK1)
047785,000053: 
047786,000054:                                                                                                  #                        C(LST1)         =       -(T2-T1)+1
047787,000055:                                                                                                  #                        C(LST1 +1)      =       -(T3-T2)+1
047788,000056:                                                                                                  #                        C(LST1 +2)      =       -(T4-T3)+1
047789,000057:                                                                                                  #                                        .
047790,000058:                                                                                                  #                                        .
047791,000059:                                                                                                  #                        C(LST1 +6)      =       -(T8-T7)+1
047792,000060:                                                                                                  #                        C(LST1 +7)      =       -(T9-T8)+1
047793,000061: 
047794,000062:                                                                                                  #                        C(LST2)         =       2CADR OF TASK1
047795,000063:                                                                                                  #                        C(LST2 +2)      =       2CADR OF TASK2
047796,000064:                                                                                                  #                                        .
047797,000065:                                                                                                  #                                        .
047798,000066:                                                                                                  #                        C(LST2 +14)     =       2CADR OF TASK8
047799,000067:                                                                                                  #                        C(LST2 +16)     =       2CADR OF TASK9
047800,000068: 
047801,000069:                                                                                                  #  WARNINGS -
047802,000070:                                                                                                  #  --------
047803,000071:                                                                                                  #        1)      1 <= C(A) <= 16250D (1 CENTISECOND TO 162.5 SEC)
047804,000072:                                                                                                  #        2)      9 TASKS MAXIMUM
047805,000073:                                                                                                  #        3)      TASKS CALLED UNDER INTERRUPT INHIBITED
047806,000074:                                                                                                  #        4)      TASKS END BY TC TASKOVER
047807,000075: 
047808,000076:                                                                                                  #  CALLING SEQUENCE -
047809,000077:                                                                                                  #        L-1     CA      DELTAT  (TIME IN CENTISECONDS TO TASK START)
047810,000078:                                                                                                  #        L       TC      WAITLIST
047811,000079:                                                                                                  #        L+1     2CADR   DESIRED TASK
047812,000080:                                                                                                  #        L+2     (MINOR OF 2CADR)
047813,000081:                                                                                                  #        L+3     RELINT          (RETURNS HERE)
047814,000082: 
047815,000083:                                                                                                  #  TWIDDLE -
047816,000084:                                                                                                  #  -------
047817,000085:                                                                                                  #        TWIDDLE IS FOR USE WHEN THE TASK BEING SET UP IS IN THE SAME EBANK AND FBANK AS THE USER.  IN
047818,000086:                                                                                                  #        SUCH CASES, IT IMPROVES UPON WAITLIST BY ELIMINATING THE NEED FOR THE BBCON HALF OF THE 2CADR,

047820,000088:                                                                                                  #        SAVING A WORD.  TWIDDLE IS LIKE WAITLIST IN EVERY RESPECT EXCEPT CALLING SEQUENCE, TO WIT-
047821,000089:                                                                                                  #                L-1     CA      DELTAT
047822,000090:                                                                                                  #                L       TC      TWIDDLE
047823,000091:                                                                                                  #                L+1     ADRES   DESIRED TASK
047824,000092:                                                                                                  #                L+2     RELINT          (RETURNS HERE)
047825,000093: 
047826,000094:                                                                                                  #  NORMAL EXIT MODES -
047827,000095:                                                                                                  #        AT L+3 OF CALLING SEQUENCE
047828,000096: 
047829,000097:                                                                                                  #  ALARM OR ABORT EXIT MODES -
047830,000098:                                                                                                  #        TC      ABORT
047831,000099:                                                                                                  #        OCT     1203    (WAITLIST OVERFLOW - TOO MANY TASKS)
047832,000100: 
047833,000101:                                                                                                  #  ERASABLE INITIALIZATION REQUIRED -
047834,000102:                                                                                                  #        ACCOMPLISHED BY FRESH START,--  LST2, ..., LST2 +16 = ENDTASK
047835,000103:                                                                                                  #                                        LST1, ..., LST1 +7  = NEG1/2
047836,000104: 
047837,000105:                                                                                                  #  OUTPUT --
047838,000106:                                                                                                  #        LST1 AND LST2 UPDATED WITH NEW TASK AND ASSOCIATED TIME.
047839,000107: 
047840,000108:                                                                                                  #  DEBRIS -
047841,000109:                                                                                                  #        CENTRALS - A,Q,L
047842,000110:                                                                                                  #        OTHER    - WAITEXIT, WAITADR, WAITTEMP, WAITBANK
047843,000111: 
047844,000112:                                                                                                  #  DETAILED ANALYSIS OF TIMING -
047845,000113:                                                                                                  #        CONTROL WILL NOT BE RETURNED TO THE SPECIFIED ADDRESS (2CADR) IN EXACTLY DELTA T CENTISECONDS.
047846,000114:                                                                                                  #        THE APPROXIMATE TIME MAY BE CALCULATED AS FOLLOWS
047847,000115:                                                                                                  #                LET TO = THE TIME OF THE TC WAITLIST
047848,000116:                                                                                                  #                LET TS = TO +147U + COUNTER INCREMENTS (SET UP TIME)
047849,000117:                                                                                                  #                LET X  = TS -(100TS)/100  (VARIANCE FROM COUNTERS)
047850,000118:                                                                                                  #                LET Y  = LENGTH OF TIME OF INHIBIT INTERRUPT AFTER T3RUPT
047851,000119:                                                                                                  #                LET Z  = LENGTH OF TIME TO PROCESS TASKS WHICH ARE DUE THIS T3RUPT BUT DISPATCHED EARLIER.
047852,000120:                                                                                                  #                         (Z=0, USUALLY)
047853,000121:                                                                                                  #                LET DELTD  = THE ACTUAL TIME TAKEN TO GIVE CONTROL TO 2CADR
047854,000122:                                                                                                  #                THEN DELTD = TS+DELTA T -X +Y +Z +1.05MS* +COUNTERS*
047855,000123:                                                                                                  #                *-THE TIME TAKEN BY WAITLIST ITSELF AND THE COUNTER TICKING DURING THIS WAITLIST TIME.
047856,000124:                                                                                                  #        IN SHORT, THE ACTUAL TIME TO RETURN CONTROL TO A 2CADR IS AUGMENTED BY THE TIME TO SET UP THE TASK:S
047857,000125:                                                                                                  #        INTERRUPT, ALL COUNTERS TICKING, THE T3RUPT PROCESSING TIME, THE WAITLIST PROCESSING TIME AND THE POSSIBILITY
047858,000126:                                                                                                  #        OF OTHER TASKS INHIBITING THE INTERRUPT.
