018583,000032: 
018584,000033:                                                                                                  #  THIS LOG SECTION CONSISTS OF PRELAUNCH ALIGNMENT AND GYRO DRIFT TESTS
018585,000034:                                                                                                  #  INTEGRATED TOGETHER TO SAVE WORDS. IT IS COMPLETELY RESTART PROOFED.
018586,000035:                                                                                                  #  THE PRELAUNCH ALIGNMENT TECHNIQUE IS BASICALLY THE SAME AS IN BLOCK 1
018587,000036:                                                                                                  #  EXCEPT THAT IT HAS BEEN SIMPLIFIED IN THE SENSE THAT SMALL ANGLE APPROX.
018588,000037:                                                                                                  #  HAVE BEEN USED. THE DRIFT TESTS USE A UNIQUE IMPLEMENTATION OF THE
018589,000038:                                                                                                  #  OPTIMUM STATISTICAL FILTER. FOR A DESCRIPTION SEE E-1973.BOTH OF THESE
018590,000039:                                                                                                  #  ROUTINES USE STANDARD SYSTEM TEST LEADIN PROCEDURES. THE INITIALIZATION
018591,000040:                                                                                                  #  PROCEDURE FOR THE DRIFT TESTS IS IN THE JDC S. THE INITIALIZATION METHOD
018592,000041: 
018593,000042:                                                                                                  #  FOR GYROCOMPASS IS IN A SOON TO COME SYSTEM TEST GROUP MEMO.
018594,000043:                                                                                                  #  THIS LOG SECTION ALWAYS STARTS BY A CADR IN IMU PERF. TESTS 2 AND BEGINS
018595,000044:                                                                                                  #  AT ESTIMS. THE PIPS ARE READ EVERY .5 SEC IN COMPASS AND 1 SEC IN DRIFT
018596,000045:                                                                                                  #  TESTS. THEN IN BOTH CASES RELEVANT COMPUTATION IS DONE. THE KEY ERASABLE
018597,000046:                                                                                                  #  IS GEOCOMPS- 0 WE ARE IN A DRIFT TEST -NONZERO WE ARE IN COMPASS.
018598,000047:                                                                                                  #  THE GYROCOMPASS HAS THE CAPABILITY TO ALIGN TO ANY ORIENTATION,HAS THE
018599,000048:                                                                                                  #  CAPABILITY TO CHANGE ORIENTATIONS WHILE RUNNING,IS COMPENSATED FOR
018600,000049:                                                                                                  #  COMPONENT ERRORS,IS CAPABLE OF OPTICAL VERIFICATION( CSM ONLY).
018601,000050:

018605,000054:                                                                                                  #  IMU COMPENSATION
018606,000055:                                                                                                  #  DURING OPTICAL VERIFICATION (CSM ONLY) ESSENTIALLY ALL OF INFLIGHT ALIGN
